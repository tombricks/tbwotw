on_actions = {
	on_startup = {
		effect = {
			every_country = {
				if = {
					limit = {
						NOT = { has_country_flag = has_starting_oob }
					}
					d_create_army = yes
				}
			}
		}
	}
	on_send_volunteers = {
		effect = {
			FROM = {
				set_country_flag = {
					flag = had_volunteers_from@FROM
					days = 3650
				} 
			}
		}
	}
}
