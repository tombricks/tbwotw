SOV_is_friend = {
	SOV_soft_aid_trigger = yes
}
SOV_is_enemy = {
	OR = {
		has_war_with = SOV
		has_country_flag = enemy_of@SOV
	}
}
SOV_soft_aid_trigger = {
	NOT = { SOV_is_enemy = yes }
	NOT = { has_government = reactionary }
	OR = {
		has_country_flag = friend_of@SOV
		has_government = pro_soviet
	}
}
SOV_volunteers_trigger = {
	NOT = { has_government = reactionary }
	OR = {
		any_country = {
			AND = {
				has_war_with = PREV
				OR = {
					has_government = pro_chinese
					SOV_is_enemy = yes
					PRC_is_friend = yes
				}
			}
		}
		AND = {
			SOV_soft_aid_trigger = yes
			enemies_strength_ratio > 0.65
		}
	}
}
