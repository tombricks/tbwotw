		SUPERPOWER_volunteers_trigger = yes
	}

	ai_strategy = {
		type = send_volunteers_desire

		type = support
		id = SOV

	}

SOV_volunteers_cheers = {
	abort_when_not_enabled = yes

	enable = {
		OR = {
			has_volunteers_amount_from = {
				tag = SOV
				count > 0
			}
			has_country_flag = had_volunteers_from@SOV
		}
	}

	ai_strategy = {
		type = befriend
		id = SOV
		value = 50
	}
}
